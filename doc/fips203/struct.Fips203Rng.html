Expand description

Struct defining the Random Bit Generator to be used.

§Example

Default RNG

let mut rng = Fips203Rng::default();

Custom RNG

use rand::{prelude::ThreadRng, Rng};

let r = rand::rng();
let f = |rng:&mut ThreadRng| {
    rng.random::<[u8;32]>()
};
let mut rng = Fips203Rng::custom(r, f);
